    </div>
    {/* Pet Display */}
    <div className="bg-gradient-to-b from-rose-900/30 to-slate-900/50 border border-rose-500/30 rounded-lg p-8 mb-8 text-center">
    <div className="text-8xl mb-4">{petState.appearance}</div>
    <h2 className="text-3xl font-bold mb-2">{petState.name}</h2>
    <p className="text-slate-400 mb-6">Level {petState.level} • {petState.personality}</p>
    {/* Stats */}
    <div className="space-y-4">
    <div>
    <div className="flex justify-between mb-1 text-sm">
    <span>好感度</span>
    <span>{petState.affection}/100</span>
    </div>
    <div className="w-full bg-slate-700 rounded-full h-4 overflow-hidden">
    <div
    className="bg-gradient-to-r from-pink-500 to-red-500 h-full transition-all"
    style={{ width: `${petState.affection}%` }}
    />
    </div>
    </div>
    </div>
    </div>
    {/* Chat */}
    <div className="bg-gradient-to-b from-slate-800/50 to-slate-900/50 rounded-lg border border-rose-500/20 p-6 h-80 overflow-y-auto mb-4 space-y-4">
    {messages.length === 0 ? (
    <div className="h-full flex items-center justify-center text-center">
    <div>
    <p className="text-slate-400 text-lg">{petState.name} との会話を重ねると...</p>
    <p className="text-slate-500 mt-2">性格と見た目が成長していきます</p>
    </div>
    </div>
    ) : (
    <>
    {messages.map((msg, idx) => (
    <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
    <div className={`rounded-lg p-4 max-w-lg text-sm ${
    msg.role === 'user'
    ? 'bg-cyan-600/50'
    : 'bg-rose-900/30 border border-rose-500/30'
    }`}>
    {msg.content}
    </div>
    </div>
    ))}
    </>
    )}
    <div ref={messagesEndRef} />
    </div>
    {/* Input */}
    <div className="flex gap-3">
    <textarea
    value={inputValue}
    onChange={(e) => setInputValue(e.target.value)}
    onKeyPress={handleKeyPress}
    placeholder={`${petState.name}に話しかけましょう...`}
    className="flex-1 px-4 py-3 rounded-lg bg-slate-800 border border-rose-500/30 text-white placeholder-slate-400 focus:outline-none focus:border-rose-500 resize-none"
    rows="3"
    />
    <button
    onClick={handleSendMessage}
    disabled={isLoading}
    className="px-6 py-3 bg-gradient-to-r from-pink-500 to-rose-500 rounded-lg font-bold self-end h-fit disabled:opacity-50"
    >
    送信
    </button>
    </div>
    </div>
    </div>
    );
    }
    // ===== CHAT MODE =====
    return (
    <div className="min-h-screen bg-gradient-to-br from-slate-950 via-slate-900 to-slate-950 text-white p-4 md:p-8">
    <div className="max-w-5xl mx-auto">
    {/* Header */}
    <div className="flex items-center justify-between mb-8 mt-4 flex-wrap gap-4">
    <div>
    <h1 className="text-4xl font-bold bg-gradient-to-r from-cyan-400 to-purple-400 bg-clip-text text-transparent">
    {currentMode.name}
    </h1>
    <p className="text-slate-300">{currentMode.description}</p>
    </div>
    <button
    onClick={() => setMode('home')}
    className="px-4 py-2 bg-slate-700 hover:bg-slate-600 rounded-lg transition-all"
    >
    ← ホーム
    </button>
    </div>
    {/* Countdown Timer */}
    {countdownActive && (
    <div className={`mb-6 p-4 rounded-lg border-2 text-center ${
    countdown <= 60
    ? 'bg-red-900/30 border-red-500 animate-pulse'
    : 'bg-slate-800/50 border-orange-500/50'
    }`}>
    <p className="text-sm text-slate-400 mb-2">この質問は</p>
    <p className="text-4xl font-bold text-white">{formatCountdown(countdown)}</p>
    <p className="text-sm text-slate-400 mt-2">後に消えてしまいます</p>
    </div>
    )}
    {/* Language Feedback */}
    {showLanguageFeedback && (
    <div className="mb-6 p-4 rounded-lg bg-blue-900/30 border border-blue-500/50 flex gap-3">
    <AlertCircle size={20} className="text-blue-400 flex-shrink-0 mt-0.5" />
    <p className="text-sm text-blue-200">{feedbackText}</p>
    </div>
    )}
    {/* Chat Messages */}
    <div className="bg-gradient-to-b from-slate-800/50 to-slate-900/50 rounded-lg border border-slate-700 p-6 h-96 overflow-y-auto mb-4 space-y-4">
    {messages.length === 0 ? (
    <div className="h-full flex items-center justify-center text-center">
    <div>
    <p className="text-slate-400 text-lg">質問や相談を入力してください</p>
    <p className="text-slate-500 mt-2">丁寧な言葉遣いを心がけると、より良い関係が築けます</p>
    </div>
    </div>
    ) : (
    <>
    {messages.map((msg, idx) => (
    <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
    <div className={`rounded-lg p-4 max-w-xl text-sm ${
    msg.role === 'user'
    ? 'bg-cyan-600/50'
    : msg.isError
    ? 'bg-red-900/30 border border-red-500/50'
    : 'bg-slate-700/50 border border-slate-600'
    }`}>
    <div className="whitespace-pre-wrap">{msg.content}</div>
    {msg.role === 'assistant' && (
    <button
    onClick={() => handleCopy(msg.content, idx)}
    className="text-xs opacity-60 hover:opacity-100 flex items-center gap-1 mt-2 transition-all"
    >
    {copied === idx ? (
    <>
    <Check size={12} /> Copied
    </>
    ) : (
    <>
    <Copy size={12} /> Copy
    </>
    )}
    </button>
    )}
    </div>
    </div>
    ))}
    {isLoading && (
    <div className="flex justify-start">
    <div className="bg-slate-700/50 rounded-lg p-4 flex items-center gap-2">
    <Loader size={18} className="animate-spin" />
    <span>考え中...</span>
    </div>
    </div>
    )}
    </>
    )}
    <div ref={messagesEndRef} />
    </div>
    {/* Input Area */}
    <div className="flex gap-3 mb-4">
    <textarea
    value={inputValue}
    onChange={(e) => setInputValue(e.target.value)}
    onKeyPress={handleKeyPress}
    placeholder="質問や相談を入力..."
    className="flex-1 px-4 py-3 rounded-lg bg-slate-800 border border-slate-700 text-white placeholder-slate-400 focus:outline-none focus:border-cyan-500 resize-none"
    rows="3"
    disabled={isLoading}
    />
    <div className="flex flex-col gap-2 self-end">
    <button
    onClick={handleSendMessage}
    disabled={isLoading || !inputValue.trim()}
    className="px-6 py-3 bg-gradient-to-r from-cyan-500 to-purple-500 rounded-lg font-bold disabled:opacity-50 flex items-center gap-2"
    >
    <Send size={18} />
    送信
    </button>
    <button
    onClick={startCountdown}
    className="px-6 py-3 bg-slate-700 hover:bg-slate-600 rounded-lg text-sm transition-all flex items-center gap-2"
    >
    <Clock size={16} />
    タイマー
    </button>
    </div>
    </div>
    {/* Tips */}
    <div className="bg-slate-800/50 border border-slate-700 rounded-lg p-4 text-sm text-slate-400">
    <p className="mb-2">💡 <span className="text-cyan-300 font-semibold">Tips:</span> 丁寧な言葉遣いや優しい表現で質問すると、より良いサポートが得られます</p>
    <p>⏰ <span className="text-orange-300 font-semibold">タイマー:</span> 「タイマー」ボタンで5分のカウントダウンが開始。その質問を大事にしましょう</p>
    </div>
    {/* Ad Modal */}
    {showAd && (
    <div className="fixed inset-0 bg-black/50 flex items-center justify-center z-50 p-4">
    <div className="bg-gradient-to-br from-slate-800 to-slate-900 rounded-lg max-w-md w-full border border-purple-500/30 p-6 relative">
    <button
    onClick={handleCloseAd}
    className="absolute top-3 right-3 p-2 hover:bg-slate-700 rounded-lg transition-all"
    title="広告を閉じる"
    >
    <X size={24} className="text-slate-400 hover:text-white" />
    </button>
    <div className="text-center space-y-4">
    <p className="text-xs text-slate-500 uppercase tracking-wider">Advertisement</p>
    <div className="bg-gradient-to-r from-cyan-600 to-purple-600 rounded-lg p-8 flex flex-col items-center justify-center min-h-40">
    <div className="text-5xl mb-3">🤖</div>
    <h3 className="text-xl font-bold mb-2">AI スキルアップ講座</h3>
    <p className="text-sm opacity-90 mb-4">次世代 AI をもっと活用しよう</p>
    <button className="bg-white text-purple-600 px-6 py-2 rounded-lg font-bold hover:bg-slate-100 transition-all text-sm">
    詳しく見る →
    </button>
    </div>
    <div className="text-xs text-slate-500 space-y-1 border-t border-slate-700 pt-3 mt-3">
    <p>✕ をクリックして広告を閉じます</p>
    <p className="text-slate-600">現在の収益: ${earnings.toFixed(3)} | 広告表示: {adClosedCount}</p>
    </div>
    </div>
    </div>
    </div>
    )}
    </div>
    </div>
    );
    }
